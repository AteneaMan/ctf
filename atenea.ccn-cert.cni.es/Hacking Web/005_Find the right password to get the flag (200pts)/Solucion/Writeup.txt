Esto descargara las fotos para despues poder decodificar cada QR, en la web tambien decia que la contrase単a cumplia con un patron de regex "^([a-z0-9_\.-]+)@([\da-z\.-]+)\.([a-z\.]{2,6})$"

import glob
import qrtools
import re


decoded = []

imagenes = glob.glob('/home/QR/85.159.210.133/35895860/5737ea2f/*.png')
for imagen in imagenes:
	decode = qrtools.QR()
	decode.decode(imagen)
	decoded.append(decode.data)

for flag in decoded:
	if re.search("^([a-z0-9_\.-]+)@([\da-z\.-]+)\.([a-z\.]{2,6})$", str(flag)):
		print str(flag)
